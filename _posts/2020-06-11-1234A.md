#include <bits/stdc++.h>

using namespace std;

int t, a, b;

int main() {
	cin >> t;
	while (t--){
		cin >> a >> b;
		if (a>b) swap(a, b);
		int diff = b-a;
		int v = max(0, (a-diff)/3);
		int ans = v*2;
		b -= v*3;
		a -= v*3;
		ans += min(a, b/2);
		cout << ans << endl;	
	}
}
